PINNs 殘差約束定義

R1 流量親和律
Q_model / Q_rated = n / n_rated
# Q ∝ n (first affinity law)
residual_R1 = |Q_pred / Q_r - hz / hz_r|
R2 功率親和律
P_model / P_rated = (n / n_rated)³
# P ∝ n³ (third affinity law)
residual_R2 = |P_pred / P_r - (hz / hz_r)^3|
R3 58.5 Hz 共振禁區
penalty_R3 = max(0, hz - 58.5)
# Hard constraint: hz ≤ 58.5 Hz
residual_R3 = penalty_R3 * w_forbidden
R4 Darcy-Weisbach 管損
η = η_ideal × (1 - k_loss × (Q/Q_r - Q_bep/Q_r)²)
# Off-BEP efficiency penalty
residual_R4 = |η_pred - η_darcy(Q, Q_bep)|
R5 空蝕邊界條件
cav_factor = clamp((hz - min_hz) / 3.0, 0, 1)
# Cavitation protection: hz ≥ min_hz + 3
residual_R5 = (1 - cav_factor) × w_cavitation
